Struct Node
{
	int data;
	Node* link;
}

#新建列表
Node* A;
A = NULL;
#Node* temp = new Node();#C++语法
Node* temp = (Node*)malloc(sizeof(struct Node));#C语法
temp->data = 2;
temp->link = NULL;
A = temp;
temp = new Node();
temp->data = 4;
temp->link = NULL;

#遍历列表
Node* temp1 =A;
while(temp1->link != NULL)
{
	temp1 = temp1->link
}

#创造连接
temp1->link = temp;
